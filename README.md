from fastapi import FastAPI
from fastapi.responses import HTMLResponse
from fastapi.middleware.cors import CORSMiddleware

app = FastAPI()

# Allow frontend calls
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

# ---------------- AI AGENTS ----------------
def manager_agent(task):
    return f"Plan: Understand {task} → Research → Generate response"

def research_agent(task):
    return f"Research: {task} is related to AI systems and automation."

def writer_agent(research):
    return f"Final Output:\n\n{research}\n\nGenerated by AI Agency 🤖"

# ---------------- API ----------------
@app.get("/run/{task}")
def run_task(task: str):
    plan = manager_agent(task)
    research = research_agent(task)
    final = writer_agent(research)

    return {
        "task": task,
        "plan": plan,
        "research": research,
        "final_output": final
    }

# ---------------- FRONTEND (HTML INSIDE PYTHON) ----------------
