from fastapi import FastAPI
from agents import manager_agent, research_agent, writer_agent

app = FastAPI()

@app.get("/")
def home():
    return {"message": "AI Agency is running"}

@app.get("/run/{task}")
def run_task(task: str):
    plan = manager_agent(task)
    research = research_agent(task)
    final = writer_agent(research)

    return {
        "task": task,
        "plan": plan,
        "research": research,
        "final_output": final
    }
def manager_agent(task):
    return f"Plan: 1. Research {task} 2. Analyze 3. Generate output"

def research_agent(task):
    return f"Research data about {task}: AI systems are used in automation, chatbots, and tools."

def writer_agent(research):
    return f"Final Report:\n\n{research}\n\nThis is generated by AI Agency system."
